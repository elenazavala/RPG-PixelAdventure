  click(x)
  {
    if(gameSettings.difficulty != (-1 + x))
    {
      for(let i = 0; i <= 2; i++)
      {
        this.buttonArr[i].setColor("white");
      }
      gameSettings.difficulty = (-1 + x);
      this.buttonArr[x].setColor("yellow");
    }
  }

  clickBack()
  {
    this.titleRef.titleMusic.stop();
    this.scene.pause();
    this.scene.start("TitleScene");
  }
